31 template <typename FunctionType = details::default_function_type,
32 typename ThreadType = std::jthread>
33 requires std::invocable<FunctionType> &&
34 std::is_same_v<void, std::invoke_result_t<FunctionType>>
35 class thread_pool {
36 public:
37 explicit thread_pool(
38 const unsigned int &number_of_threads = std::thread::hardware_concurrency())
39 : tasks_(number_of_threads) {
40 std::size_t current_id = 0;
41 for (std::size_t i = 0; i < number_of_threads; ++i) {
42 try {
43 threads_.emplace_back([&, id = current_id](const std::stop_token &stop_tok) {
44 do {
45 // wait until signaled
46 tasks_[id].signal.acquire();
47
48 do {
49 // invoke the task
50 while (auto task = tasks_[id].tasks.pop()) {
51 try {
52 pending_tasks_.fetch_sub(1, std::memory_order_release);
53 std::invoke(std::move(task.value()));
54 } catch (...) {
55 }
56 }
57
58 // try to steal a task
59 for (std::size_t j = 1; j < tasks_.size(); ++j) {
60 const std::size_t index = (id + j) % tasks_.size();
61 if (auto task = tasks_[index].tasks.steal()) {
62 // steal a task
63 pending_tasks_.fetch_sub(1, std::memory_order_release);
64 std::invoke(std::move(task.value()));
65 // stop stealing once we have invoked a stolen task
66 break;
67 }
68 }
69
70 } while (pending_tasks_.load(std::memory_order_acquire) > 0);
71
72 } while (!stop_tok.stop_requested());
73 });
74 // increment the thread id
75 ++current_id;
76
77 } catch (...) {
78 // catch all
79
80 // remove one item from the tasks
81 tasks_.pop_back();
82 }
83 }
84 }

112 [[nodiscard]] std::future<ReturnType> enqueue(Function f, Args... args) {
113#if __cpp_lib_move_only_function
114 // we can do this in C++23 because we now have support for move only functions
115 std::promise<ReturnType> promise;
116 auto future = promise.get_future();
117 auto task = [func = std::move(f), ... largs = std::move(args),
118 promise = std::move(promise)]() mutable {
119 try {
120 if constexpr (std::is_same_v<ReturnType, void>) {
121 func(largs...);
122 promise.set_value();
123 } else {
124 promise.set_value(func(largs...));
125 }
126 } catch (...) {
127 promise.set_exception(std::current_exception());
128 }
129 };
130 enqueue_task(std::move(task));
131 return future;
132#else
133 /*
134 * use shared promise here so that we don't break the promise later (until C++23)
135 *
136 * with C++23 we can do the following:
137 *
138 * std::promise<ReturnType> promise;
139 * auto future = promise.get_future();
140 * auto task = [func = std::move(f), ...largs = std::move(args),
141 promise = std::move(promise)]() mutable {...};
142 */
143 auto shared_promise = std::make_shared<std::promise<ReturnType>>();
144 auto task = [func = std::move(f), ... largs = std::move(args),
145 promise = shared_promise]() {
146 try {
147 if constexpr (std::is_same_v<ReturnType, void>) {
148 func(largs...);
149 promise->set_value();
150 } else {
151 promise->set_value(func(largs...));
152 }
153
154 } catch (...) {
155 promise->set_exception(std::current_exception());
156 }
157 };
158
159 // get the future before enqueuing the task
160 auto future = shared_promise->get_future();
161 // enqueue the task
162 enqueue_task(std::move(task));
163 return future;
164#endif
